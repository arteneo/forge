# Basic usage

```jsx
import { FormikValues } from "formik";
import { FieldsInterface, Email, Text } from "@arteneo/forge";

const BasicForm = () => {
    const onSubmit = (values: FormikValues) => {
        // You should fill your submit logic here
        console.log(values);
    };

    const fields: FieldsInterface = {
        firstName: <Text />,
        lastName: <Text />,
        email: <Email />,
    };

    return <Form {...{fields, onSubmit}} />;
};

export default BasicForm;
```
